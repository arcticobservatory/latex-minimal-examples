\section{With a custom macro}

% Adapted from https://tex.stackexchange.com/a/35936
\renewcommand{\epigraph}[2]{
    \hfill\begin{minipage}{\dimexpr\textwidth-1cm}%
    \begin{quotation}%
        #1%
    \end{quotation}%
    \begin{flushright}%
        --- #2%
    \end{flushright}%
    \end{minipage}%
}

  an oppressive cyberpunk dystopia. Here you go.}
